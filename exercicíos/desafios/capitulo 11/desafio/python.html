num = int(input('Qual número: '))
if num % 2 == 0:
    print(f'O número {num} é par')
else if num % 2 == 1:
    print('O número {} é ímpar'.format(num))
else:
    print('Não é um número')

def fatorial(n):
    f = 1
    while n > 0:
        f *= n
        n -= 1
    return f


def anagramas(palavra):
    #verifica se repete e conta os repetidos e nao repetidos
    repete = []
    nao_repete = []
    for letra in palavra.strip():
        if letra not in nao_repete:
            nao_repete.append(letra)
        elif letra in nao_repete:
            repete.append(letra)
    #calculos
    #sem repetir
    if not repete:
        resultado = fatorial(len(nao_repete))

        # repetindo
    else:
        fatorial_cima = fatorial(len(repete) + len(nao_repete))# vai modificar depois, preciso usa-ló agora
        fatorial_baixo = 1#parte de baixo x1! * x2! * xn!

        for c in range(len(repete)):
            #quando acabar, inves de erro ele quebra o laço
            try:
                letra = repete[0]
            except:
                break
            fatorial_baixo *= fatorial(repete.count(letra) + 1)#multiplica direto os de baixo
            while letra in repete:#tira os repetidos já usados
                repete.remove(letra)
        #calculo do anagrama, int() só pra ficar bonito
        resultado = int(fatorial_cima / fatorial_baixo)


    return  resultado
